import java.io.*;
import java.util.*;

public class Main {

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(br.readLine());
        int[] arr = new int[n];
        Set<Integer> set = new HashSet<>(); // ì œê±°í•  ìˆ«ì ì§‘í•©

        for(int i=0; i<n; i++) {
            int x = Integer.parseInt(br.readLine());
            arr[i] = x;
            set.add(x);
        }


        int ans = 1; // ì—°ì† ìµœì¥ ê¸¸ì´

        for(int k:set){
            int cnt = 1; // í˜„ì¬ ì—°ì† ê¸¸ì´
            int prev = -1;

            for(int i=0; i<n; i++){
                if(arr[i] == k) continue; // k ì œê±°(ê±´ë„ˆë›°ê¸°)

                if(prev == arr[i]) {
                    cnt++;
                    ans = Math.max(ans, cnt);
                }else cnt = 1;

                prev = arr[i];
            }

        }

        System.out.println(ans);
    }
}
