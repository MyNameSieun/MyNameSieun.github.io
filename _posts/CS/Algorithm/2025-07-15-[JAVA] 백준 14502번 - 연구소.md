import java.io.*;
import java.util.*;

public class Main {
    static int[][] board;
    static int[][] copyBoard;

    static int n, m;
    static int maxSafe = 0;

    public static void main (String[]args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        // ì…ë ¥
        StringTokenizer st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken()); // ì—´
        m = Integer.parseInt(st.nextToken()); // í–‰

        board = new int[n][m];
        copyBoard = new int[n][m];

        // ì´ˆê¸°í™”
        for(int i=0; i<n; i++){
            st = new StringTokenizer(br.readLine());
            for(int j=0; j<m; j++){
                board[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        buildWall(0);
        System.out.println(maxSafe);
    }

    // ë²½ ì„¸ìš°ê¸° (ì¡°ê±´: map[i][j]=0ì´ì—¬ì•¼í•¨, ë¬´ì¡°ê±´ 3ê°œ)
    static void buildWall(int cnt){

        }
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(board[i][j] == 0){
                    board[i][j] = 1;
                    buildWall(cnt + 1);
                    board[i][j] = 0; // ë°±íŠ¸ë˜í‚¹
                }
            }
        }
    }

    // ë°”ì´ëŸ¬ìŠ¤ í¼íŠ¸ë¦¬ê¸°
    static void spreadVirus(){
        for(int i=0; i<n; i++){
            copyBoard[i] = board[i].clone(); // ë°°ì—´ í•œ ì¤„ì”© ë³µì‚¬ (ê¹Šì€ë³µì‚¬)
        }

            for(int j=0; j<m; j++){

                    dq.offer(new int[]{i, j});
                }
            }
        }

        int[] dx = {-1, 1, 0, 0};
        int[] dy = {0, 0, -1, 1};

        while (!dq.isEmpty()){
            int[] cur = dq.poll();
            int cx = cur[0];
            int cy = cur[1];

            for(int i=0; i<4; i++){
                int nx = cx + dx[i];
                int ny = cy + dy[i];

                if(nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
                if(copyBoard[nx][ny] == 0){
                    copyBoard[nx][ny] = 2;
                    dq.offer(new int[]{nx, ny});
                }
            }
        }

        int safe = 0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(copyBoard[i][j] == 0) safe++;
            }
        }

        maxSafe = Math.max(maxSafe, safe);

    }
}
