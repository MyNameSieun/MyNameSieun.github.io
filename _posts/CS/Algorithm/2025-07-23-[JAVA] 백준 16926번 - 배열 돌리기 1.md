import java.io.*;
import java.util.*;

public class Main {
    static int n;
    static int m;
    static int r;
    static int[][] map;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        // ì…ë ¥
        StringTokenizer st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        r = Integer.parseInt(st.nextToken());
        map = new int[n][m];

        // ì´ˆê¸°í™”

            }

            }

                map[k][m- i -1] = map[k+1][m- i -1];
            }

            }

    static void rotate() {
        // ì¢Œ, í•˜, ìš°, ìƒ
        int[] dx = {0, 1, 0, -1};
        int[] dy = {1, 0, -1, 0};

        for (int i = 0; i < Math.min(n, m) / 2; i++) {
            int x = i;
            int y = i;
            int temp = map[x][y];

            int height = n - 2 * i;
            int width = m - 2 * i;
            int len = 2 * (height + width) - 4;

            int dir = 0;

            for (int step = 0; step < len; step++) {
                int nx = x + dx[dir];
                int ny = y + dy[dir];

                // ë²”ìœ„ ë²—ì–´ë‚˜ë©´ ë°©í–¥ ì „í™˜
                if (nx < i || nx >= n - i || ny < i || ny >= m - i) {
                    dir++;
                    nx = x + dx[dir];
                    ny = y + dy[dir];
                }

                map[x][y] = map[nx][ny];
                x = nx;
                y = ny;
            }

            map[i + 1][i] = temp;
        }
    }
