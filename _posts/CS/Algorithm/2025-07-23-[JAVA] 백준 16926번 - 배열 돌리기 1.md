# 💻 전체 코드

## 방향 벡터 미사용

레이어 회전은 항상 좌 -> 하 -> 우 -> 상 순서로 움직이기 때문에 방향 벡터를 사용하지 않는 것이 더 깔끔한 것 같다.

```java
import java.io.*;
import java.util.*;

public class Main {
    static int n;
    static int m;
    static int r;
    static int[][] map;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        // 입력
        StringTokenizer st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        r = Integer.parseInt(st.nextToken());
        map = new int[n][m];

        // 초기화
        for(int i=0; i<n; i++){
            st = new StringTokenizer(br.readLine());
            for(int j=0; j<m; j++){
                map[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        // r번 회전
        while(r -- > 0){
            rotate();
        }
        printArr();
    }

    // 배열 반시계 방향으로 한 칸 회전
    static void rotate(){
        // 각 레이어 개별적으로 방문하며 회전
        for(int i = 0; i < Math.min(n, m)/2; i++) {

            // 왼쪽 위 값 저장 -> 맨 마지막에 복원해야 함
            int temp = map[i][i];

            // 좌
            for(int k = i; k<m- i -1; k++) {
                map[i][k] = map[i][k+1];
            }

            // 상
            for(int k = i; k<n-1- i; k++) {
                map[k][m- i -1] = map[k+1][m- i -1];
            }

            // 우
            for(int k = m- i -1; k> i; k--) {
                map[n-1- i][k] = map[n-1- i][k-1];
            }

            // 좌
            for(int k = n- i -1; k> i; k--) {
                map[k][i] = map[k-1][i];
            }
            map[i +1][i] = temp;
        }
    }

    // 출력
    static void printArr(){
        StringBuilder sb = new StringBuilder();

        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                sb.append(map[i][j]).append(" ");
            }
            sb.append("\n");
        }
        System.out.println(sb);
    }
}
```

## 방향 벡터 사용

```java
    static void rotate() {
        // 좌, 하, 우, 상
        int[] dx = {0, 1, 0, -1};
        int[] dy = {1, 0, -1, 0};

        for (int i = 0; i < Math.min(n, m) / 2; i++) {
            int x = i;
            int y = i;
            int temp = map[x][y];

            int height = n - 2 * i;
            int width = m - 2 * i;
            int len = 2 * (height + width) - 4;

            int dir = 0;

            for (int step = 0; step < len; step++) {
                int nx = x + dx[dir];
                int ny = y + dy[dir];

                // 범위 벗어나면 방향 전환
                if (nx < i || nx >= n - i || ny < i || ny >= m - i) {
                    dir++;
                    nx = x + dx[dir];
                    ny = y + dy[dir];
                }

                map[x][y] = map[nx][ny];
                x = nx;
                y = ny;
            }

            map[i + 1][i] = temp;
        }
    }
```
