import java.io.*;

public class Main {
    static int[][] arr;
    static int n;
    static int colAns = 0, rowAns = 0;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        n = Integer.parseInt(br.readLine());
        arr = new int[n][n];

        for(int i=0; i<n; i++){
            String line = br.readLine();
            for(int j=0; j<n; j++){
                arr[i][j] = line.charAt(j);
            }
        }

        solve();
        System.out.println(rowAns + " " + colAns);

    }

    static void solve(){
        // í–‰ ì²´í¬
        for(int i=0; i<n; i++){
            int rowCnt = 0;
            for(int j=0; j<n; j++){
                if(arr[i][j] == '.') {
                    rowCnt++;
                }else if(arr[i][j] == 'X'){
                    if(rowCnt >=2) rowAns ++;
                    rowCnt = 0;
                }
            }

            if(rowCnt >= 2) rowAns++; // í–‰ì˜ ë í™•ì¸

                }
            }
