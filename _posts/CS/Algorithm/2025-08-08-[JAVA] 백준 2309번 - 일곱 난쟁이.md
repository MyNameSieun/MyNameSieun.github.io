import java.io.*;
import java.util.*;

public class Main {
    static int[] arr;
    static int sum;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        arr = new int[9];

        sum = 0;
        for(int i=0; i<9; i++){
            arr[i] = Integer.parseInt(br.readLine());
            sum += arr[i];
        }

        solve();
        Arrays.sort(arr);
        for(int i=0; i<9; i++){
            if(arr[i] != 0){
                System.out.println(arr[i]);
            }
        }
    }
    static void solve(){
        for(int i=0; i<8; i++) {
            for (int j = i + 1; j < 9; j++) {
                int temp = arr[i] + arr[j];
                if (sum - temp == 100) {
                    arr[i] = 0;
                    arr[j] = 0;
                    return;
                }
            }
        }
    }
}
