import java.io.*;
import java.util.*;

public class Main {
    static int[] board;
    static boolean[] visited;
    static int[] dist;

    static int bfs(int start){
        Deque<Integer> dq = new ArrayDeque<>();
        dq.offer(start);
        visited[start] = true;

        while (!dq.isEmpty()){
            int current = dq.poll();
            for(int i=1; i<=6; i++){
                int next = current + i;

                if(next > 100) continue;
                int move = board[next];

                if (!visited[move]) {
                    dq.offer(move);
                    visited[move] = true;
                    dist[move] = dist[current] + 1;
                }
            }
        }
        return dist[100];

    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
