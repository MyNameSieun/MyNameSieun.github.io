import java.io.*;
import java.util.*;

public class Main {
    static int[][] arr;
    static boolean[][] v;
    static int n;
    static int ans = Integer.MAX_VALUE;

    static int[] dx = {-1, 1, 0, 0};
    static int[] dy = {0, 0, -1, 1};

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = Integer.parseInt(br.readLine());
        arr = new int[n][n];
        v = new boolean[n][n];

        for (int i = 0; i < n; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            for (int j = 0; j < n; j++) {
                arr[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        dfs(0, 0);
        System.out.println(ans);
    }

    static void dfs(int depth, int sum){
        // 1. ì¢…ë£Œì¡°ê±´
        if(depth == 3){
            ans = Math.min(ans, sum);
            return;
        }

        // 2. íƒìƒ‰
        for(int i=1; i<n-1; i++){
            for(int j=1; j<n-1; j++){
                if(isValid(i, j)){ // 5ì¹¸ ê°€ëŠ¥ í™•ì¸
                    int cost = getCost(i,j); // 5ì¹¸ ë¹„ìš© ê³„ì‚°

                    setVisited(i, j, true); // 5ì¹¸ ë°©ë¬¸ ì²˜ë¦¬
                    dfs(depth + 1, sum + cost);
                    setVisited(i, j, false); // 5ì¹¸ ë°©ë¬¸ í•´ì œ
                }

            }
        }
    }

    // 5ì¹¸ ê°€ëŠ¥í•œì§€ í™•ì¸
    static boolean isValid(int x, int y){
        if(v[x][y]) return false;
        for(int d=0; d<4; d++){
            int nx = x + dx[d];
            int ny = y + dy[d];

            if(v[nx][ny]) return false;
        }

        return true;
    }

    // 5ì¹¸ ë¹„ìš© ê³„ì‚°
    static int getCost(int x, int y){
        int cost = arr[x][y];

        for(int d=0; d<4; d++){
            int nx = x + dx[d];
            int ny = y + dy[d];

            cost += arr[nx][ny];
        }

        return cost;
    }

    // 5ì¹¸ ë°©ë¬¸ ì²˜ë¦¬ ë° ë°±íŠ¸ë˜í‚¹
    static void setVisited(int x, int y, boolean flag){
        v[x][y] = flag;
        for(int d=0; d<4; d++){
            int nx = x + dx[d];
            int ny = y + dy[d];

            v[nx][ny] = flag;
        }
    }
}
