```java
import java.io.*;

public class Solution {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        int T = 10;
        for(int tc=1; tc<=T; tc++) {
            int t = Integer.parseInt(br.readLine());

            // 입력 및 초기화
            char[][] arr = new char[100][100];

            for (int i = 0; i < 100; i++) {
                String line = br.readLine();
                for (int j = 0; j < 100; j++) {
                    arr[i][j] = line.charAt(j);
                }
            }

            int max = 0;  // 가장 긴 회문 길이

            // 회문 탐색 (길이 100부터 1까지 줄여가며 검사)
            for (int len = 100; len >= 1; len--) {
                boolean found = false;

                // 가로 회문 탐색
                for (int i = 0; i < 100; i++) {
                    for (int start = 0; start <= 100 - len; start++) {
                        StringBuilder sb = new StringBuilder();
                        for (int j = start; j < start + len; j++) {
                            sb.append(arr[i][j]);
                        }
                        if (isPalindrome(sb)) {
                            max = len;
                            found = true;
                            break;
                        }
                    }
                }

                // 세로 회문 탐색
                    for (int i = 0; i < 100; i++) {
                        for (int start = 0; start <= 100 - len; start++) {
                            StringBuilder sb = new StringBuilder();
                            for (int j = start; j < start + len; j++) {
                                sb.append(arr[j][i]);
                            }
                            if (isPalindrome(sb)) {
                                max = len;
                                found = true; // 세로에서 회문을 찾았으므로
                                break;
                            }
                        }
                }
                if (found) break;
            }


            // 출력
            System.out.print("#" + t + " ");
            System.out.println(max);
        }
    }
    private static boolean isPalindrome(StringBuilder sb) {
        String str = sb.toString();
        String rstr = sb.reverse().toString();
        if (str.equals(rstr)) {
            return true;
        }
        return false;
    }
}
```
