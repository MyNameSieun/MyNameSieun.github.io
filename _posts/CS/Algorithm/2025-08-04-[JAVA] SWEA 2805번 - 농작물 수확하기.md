import java.io.*;
import java.util.*;

public class Solution {
    static int n, result;
    static int[][] form;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        int t = Integer.parseInt(br.readLine());
        for(int tc=1; tc<=t; tc++){

            // ì…ë ¥ ë° ì´ˆê¸°í™”
            n = Integer.parseInt(br.readLine()); // ë†ì¥ì˜ í¬ê¸°
            form = new int[n][n];
            for(int i=0; i<n; i++){
                String line = br.readLine();
                for(int j=0; j<n; j++){
                    char ch = line.charAt(j);
                    form[i][j] = ch - '0';
                }
            }

            solved();
            print(tc);

        }
    }
    static void solved(){
        result = 0;
        int m = n/2;

        for(int i=0; i<n; i++) {
            // ìœ—ë¶€ë¶„
            if(i <= m){
                for(int j=m-i; j<=m+i; j++){
                    result += form[i][j];
                }
            }
            // ì•„ë«ë¶€ë¶„
            else{
                for(int j=i-m; j<n-(i-m); j++){
                    result += form[i][j];
                }
            }
        }
    }

    static void print(int tc){
        // ì¶œë ¥
        StringBuilder sb = new StringBuilder();
        sb.append("#").append(tc).append(" ");
        sb.append(result);
        System.out.println(sb);
    }
}

static void solved() {
    result = 0;
    int m = n / 2;
    int start = m, end = m;

    for (int i = 0; i < n; i++) {
        for (int j = start; j <= end; j++) {
            result += form[i][j];
        }

        if (i < m) {
            start--;
            end++;
        } else {
            start++;
            end--;
        }
    }
}
