# 💻 전체 코드

클래스 생성하는 것에 더 익숙해져야 할 것 같다.

```java
import java.io.*;
import java.util.*;

public class Main {
    static int money;
    static int[] stockPrice;

    static class Stock {
        int money = 0; // 현재 가진 돈
        int cnt = 0; // 현재 가진 주식 수

        public Stock(int money, int cnt) {
            this.money = money;
            this.cnt = cnt;
        }

        public void buy_stock(int cnt, int todayCost) {
            money -= todayCost * cnt;
            this.cnt += cnt;
        }

        public void sell(int todayCost) {
            money += this.cnt * todayCost;
            this.cnt = 0;
        }
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        money = Integer.parseInt(br.readLine());
        stockPrice = new int[14];

        StringTokenizer st = new StringTokenizer(br.readLine());
        for (int i = 0; i < 14; i++) {
            stockPrice[i] = Integer.parseInt(st.nextToken());
        }

        int bnp = BNP();
        int timing = TIMING();

        if (bnp < timing)
            System.out.println("TIMING");
        else if (timing < bnp)
            System.out.println("BNP");
        else
            System.out.println("SAMESAME");
    }

    // 준현
    static int BNP() {
        Stock s = new Stock(money, 0);

        for (int i = 0; i < 14; i++) {
            if (stockPrice[i] <= s.money) {
                int canBuy = s.money / stockPrice[i];
                s.buy_stock(canBuy, stockPrice[i]);
            }
        }

        return s.money + (s.cnt * stockPrice[13]);
    }

    // 성민
    static int TIMING() {
        Stock s = new Stock(money, 0);

        for (int i = 3; i < 14; i++) {
            if (stockPrice[i - 1] < stockPrice[i - 2] && stockPrice[i - 2] < stockPrice[i - 3]) {
                int canBuy = s.money / stockPrice[i];
                s.buy_stock(canBuy, stockPrice[i]);

            } else if (stockPrice[i - 1] > stockPrice[i - 2] && stockPrice[i - 2] > stockPrice[i - 3]) {
                s.sell(stockPrice[i]);
            }
        }

        return s.money + (s.cnt * stockPrice[13]);
    }
}
```
