import java.io.*;
import java.util.*;

public class Main {
    // ìƒìš°í•˜ì¢Œ
    static int[] dx = {-1, 0, 1, 0};
    static int[] dy = {0, 1, 0, -1};

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(br.readLine());
        String str = br.readLine();

        char[][] arr = new char[101][101];

        int dir = 2; // ë‚¨ìª½(í•˜)

        int x = 50, y = 50;
        int minRow = 50, maxRow = 50, minCol = 50, maxCol = 50;

        arr[x][y] = '.';

        for (int i = 0; i < n; i++) {
            char ch = str.charAt(i);
            if (ch == 'F') { // ì•ìœ¼ë¡œ í•œ ì¹¸
                x += dx[dir];
                y += dy[dir];
                arr[x][y] = '.';

                minRow = Math.min(minRow, x);
                maxRow = Math.max(maxRow, x);
                minCol = Math.min(minCol, y);
                maxCol = Math.max(maxCol, y);
            } else if (ch == 'L') dir = (dir - 1 + 4) % 4; // ì™¼ìª½ìœ¼ë¡œ íšŒì „
            else if (ch == 'R') dir = (dir + 1) % 4; // ì˜¤ë¥¸ìª½ìœ¼ë¡œ íšŒì „
        }

        for(int i=minRow; i<=maxRow; i++) {
            for(int j=minCol; j<=maxCol; j++){
                if(arr[i][j] != '.') arr[i][j] = '#';
                System.out.print(arr[i][j]);
            }
            System.out.println();
        }
    }
}
